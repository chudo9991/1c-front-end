7.3.1 Глобальные переменные

Попробуем связать числа в компоненатах Enlarger и Reducer. Для этого создадим модуль globals.js, экспортирующий словать с полем value.

Переписываем компоненты Enlarger и Reducer с использованием переменной value.

        Листинг 7.22 globals.js
        
        module.exports={
            value:0
        };
        
        Листинг 7.23 enlarger.vue

        <template>
            <h1 v-on:mouseover="enlarge">{{ referenced.value }}</h1>
        </template>
        <script>
            let globals=require("./globals.js");
        
            module.exports={
                data: function() {
                    return {referenced: globals};
                },
                methods: {
                    enlarge: function() {
                        this.referenced.value+=1;
                    }
                }
            };
        </script>
        
        Листинг 7.23 reducer.vue

        <template>
            <button type="button" v-on:click="reduce">{{ referenced.value }}</button>
        </template>
        <script>
            let globals=require("./globals.js");
            module.exports={
                data: function() {
                    return {referenced: globals};
                },
            methods: {
                reduce: function() {
                    this.referenced.value-=1;
                }
            }
        };
        </script>
